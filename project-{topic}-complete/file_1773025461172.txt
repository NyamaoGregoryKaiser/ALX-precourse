import React, { useState, useEffect } from 'react';
import axios from '../api/axios';
import { useAuth } from '../contexts/AuthContext'; // To get current user and check roles

function UsersPage() {
  const [users, setUsers] = useState([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState('');
  const [currentPage, setCurrentPage] = useState(1);
  const [totalPages, setTotalPages] = useState(1);
  const { user: currentUser } = useAuth(); // Get currently logged-in user
  const usersPerPage = 10;

  useEffect(() => {
    fetchUsers();
  }, [currentPage]);

  const fetchUsers = async () => {
    setLoading(true);
    setError('');
    try {
      const response = await axios.get(`/users`, {
        params: {
          page: currentPage,
          limit: usersPerPage,
          sortBy: 'createdAt:desc'
        }
      });
      setUsers(response.data.results);
      setTotalPages(response.data.totalPages);
    } catch (err) {
      setError(err.response?.data?.message || 'Failed to fetch users.');
    } finally {
      setLoading(false);
    }
  };

  const handleUpdateRole = async (userId, newRole) => {
    if (!window.confirm(`Are you sure you want to change the role of this user to ${newRole}?`)) return;
    setError('');
    try {
      await axios.patch(`/users/${userId}`, { role: newRole });
      fetchUsers(); // Re-fetch users to show updated role
    } catch (err) {
      setError(err.response?.data?.message || 'Failed to update user role.');
    }
  };

  const handleDeleteUser = async (userId) => {
    if (!window.confirm('Are you sure you want to delete this user? This action cannot be undone.')) return;
    setError('');
    try {
      await axios.delete(`/users/${userId}`);
      fetchUsers(); // Re-fetch users after deletion
    } catch (err) {
      setError(err.response?.data?.message || 'Failed to delete user.');
    }
  };

  if (loading) return <p>Loading users...</p>;
  if (error) return <p className="alert alert-error">{error}</p>;

  const availableRoles = ['user', 'editor', 'admin']; // Define roles

  return (
    <div className="container">
      <div className="page-header">
        <h1>Users</h1>
      </div>

      {users.length === 0 ? (
        <p>No users found.</p>
      ) : (
        <>
          <table className="data-table">
            <thead>
              <tr>
                <th>Name</th>
                <th>Email</th>
                <th>Role</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {users.map((user) => (
                <tr key={user.id}>
                  <td data-label="Name">{user.name}</td>
                  <td data-label="Email">{user.email}</td>
                  <td data-label="Role">
                    {/* Only allow changing role if not the current user and current user is admin */}
                    {currentUser?.role === 'admin' && currentUser.id !== user.id ? (
                      <select
                        value={user.role}
                        onChange={(e) => handleUpdateRole(user.id, e.target.value)}
                      >
                        {availableRoles.map((role) => (
                          <option key={role} value={role}>{role}</option>
                        ))}
                      </select>
                    ) : (
                      user.role
                    )}
                  </td>
                  <td className="actions">
                    {/* Only allow deleting if not the current user and current user is admin */}
                    {currentUser?.role === 'admin' && currentUser.id !== user.id && (
                      <button onClick={() => handleDeleteUser(user.id)} className="delete-btn">Delete</button>
                    )}
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
          <div className="pagination">
            <button onClick={() => setCurrentPage(prev => Math.max(prev - 1, 1))} disabled={currentPage === 1}>
              Previous
            </button>
            <span>Page {currentPage} of {totalPages}</span>
            <button onClick={() => setCurrentPage(prev => Math.min(prev + 1, totalPages))} disabled={currentPage === totalPages}>
              Next
            </button>
          </div>
        </>
      )}
    </div>
  );
}

export default UsersPage;
